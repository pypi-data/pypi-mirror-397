mutation containerFqdnAddValues ( $accountId:ID! $fqdnContainerAddValuesInput:FqdnContainerAddValuesInput!) {
	container ( accountId:$accountId ) {
		fqdn {
			addValues ( input:$fqdnContainerAddValuesInput   ) {
				container {
					id
					name
					description
					size
					audit {
						createdBy
						lastModifiedBy
						lastModifiedAt
						createdAt
					}
				}
			}
		}
	}	
}
