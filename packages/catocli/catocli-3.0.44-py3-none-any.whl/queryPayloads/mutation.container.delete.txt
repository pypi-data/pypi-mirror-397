mutation containerDelete ( $accountId:ID! $deleteContainerInput:DeleteContainerInput!) {
	container ( accountId:$accountId ) {
		delete ( input:$deleteContainerInput  )  {
			container {
				id
				name
				description
				size
				audit {
					createdBy
					lastModifiedBy
					lastModifiedAt
					createdAt
				}
			}
		}
	}	
}
